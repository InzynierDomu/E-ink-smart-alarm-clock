// Ustaw piny zgodnie z dokumentacją CrowPanel 5.79"
#define CS 15  // Chip Select
#define DC 2   // Data/Command
#define RST 17 // Reset
#define BUSY 4 // Busy
// Wybór sterownika zbliżonego do SSD1683 272x792 (najbliższe:
// GxEPD2_790c_SSD1683)
GxEPD2_BW<GxEPD2_579_GDEY0579T93, GxEPD2_579_GDEY0579T93::HEIGHT> display(
    GxEPD2_579_GDEY0579T93(/*CS=5*/ 45, /*DC=*/46, /*RST=*/47, /*BUSY=*/48));
#define GxEPD2_DISPLAY_CLASS GxEPD2_BW
#define GxEPD2_DRIVER_CLASS                                                    \
  GxEPD2_579_GDEY0579T93 // GDEY0579T93 792x272, SSD1683 (FPC-E004 22.04.13)
#define SCR_WIDTH 792
#define SCR_HEIGHT 272
#define LVBUF ((SCR_WIDTH * SCR_HEIGHT / 8) + 8) // increased buffer
bool screen; // flags for data
static lv_display_t *lvDisplay;
static uint8_t lvBuffer[2][LVBUF];
void my_disp_flush(lv_display_t *disp, const lv_area_t *area,
                   unsigned char *data) {
  int16_t width = area->x2 - area->x1 + 1;
  int16_t height = area->y2 - area->y1 + 1;
  display.drawImage((uint8_t *)data + 8, area->x1, area->y1, width, height);
  lv_display_flush_ready(disp);
  // flushing complete when bottom right area flushed
  screen = (area->x1 + width == SCR_WIDTH) && (area->y1 + height == SCR_HEIGHT);
}
static uint32_t my_tick(void) { return millis(); }

}
